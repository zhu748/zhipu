    authToken=key;
    try{localStorage.setItem('zcodeProxyAuth',key)}catch{}

    authToken=saved;
    document.getElementById('loginWrap').style.display='none';
    document.getElementById('app').classList.add('show');
    initDashboard();
  }catch{}
})();

    server:{port:+document.getElementById('cfgPort').value,host:document.getElementById('cfgHost').value},
    provider:document.getElementById('cfgProvider').value,
    plan:document.getElementById('cfgPlan').value,
    auth:{mode:document.getElementById('cfgAuthMode').value},
    defaultModel:document.getElementById('cfgDefaultModel').value,
    models:document.getElementById('cfgModels').value.split('\n').map(s=>s.trim()).filter(Boolean),
    retry:{maxRetries:+document.getElementById('cfgRetryMax').value,initialDelayMs:+document.getElementById('cfgRetryDelay').value,maxDelayMs:+document.getElementById('cfgRetryMaxDelay').value,backoffFactor:+document.getElementById('cfgRetryBackoff').value,retryableStatuses:document.getElementById('cfgRetryStatuses').value.split(',').map(s=>parseInt(s.trim())).filter(n=>n>0)},

async function startOAuth(){
  const provider=document.getElementById('oauthProvider').value;
  const plan=document.getElementById('oauthPlan').value;
  document.getElementById('oauthFlowStatus').style.display='block';
  document.getElementById('oauthDot').className='oauth-dot off';
  document.getElementById('oauthStatusText').textContent='初始化中...';
  document.getElementById('oauthUrlBox').style.display='none';
  document.getElementById('callbackResult').style.display='none';
  try{
    const r=await fetch(API+'/admin/api/oauth/init',{method:'POST',headers:authHeaders(),body:JSON.stringify({provider,plan})});
    const d=await r.json();
    if(!r.ok)throw new Error(d.error||'Init failed');
    currentFlowId=d.flowId;
    sessionStorage.setItem('zcodeOAuthFlowId',d.flowId);
    document.getElementById('oauthDot').className='oauth-dot on';
    document.getElementById('oauthStatusText').textContent='等待授权中...';
    document.getElementById('oauthUrlBox').style.display='block';
    document.getElementById('oauthUrl').value=d.authorizeUrl;
    oauthExpiresAt=d.expiresAt||0;
    startOauthCountdown();
    pollOAuth(d.flowId);
  }catch(e){toast(e.message,'error');document.getElementById('oauthStatusText').textContent='失败：'+e.message}
}
function startOauthCountdown(){
  if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
  const update=()=>{
    const remainingMs=oauthExpiresAt?oauthExpiresAt-Date.now():0;
    if(remainingMs<=0){
      document.getElementById('oauthCountdown').textContent='已过期';
      clearInterval(oauthExpiryTimer);
      return;
    }
    const s=Math.floor(remainingMs/1000);
    const m=Math.floor(s/60),ss=s%60;
    document.getElementById('oauthCountdown').textContent=`${m}:${String(ss).padStart(2,'0')}`;
  };
  update();
  oauthExpiryTimer=setInterval(update,1000);
}
async function pollOAuth(flowId){
  for(let i=0;i<120;i++){
    await new Promise(r=>setTimeout(r,2000));
    try{
      const r=await fetch(API+'/admin/api/oauth/poll?flowId='+flowId,{headers:authHeaders()});
      const d=await r.json();
      if(d.status==='ready'){
        document.getElementById('oauthDot').className='oauth-dot on';
        document.getElementById('oauthStatusText').innerHTML='<span class="badge badge-success"><span class="badge-dot"></span>已授权！</span>';
        document.getElementById('oauthUrlBox').style.display='none';
        if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
        toast('OAuth 登录成功','success');
        currentFlowId=null;
        sessionStorage.removeItem('zcodeOAuthFlowId');
        loadAccountInfo();loadCredStore();
        return;
      }
      if(d.status==='failed'){
        document.getElementById('oauthDot').className='oauth-dot off';
        document.getElementById('oauthStatusText').textContent='授权失败';
        if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
        toast('OAuth 失败','error');return;
      }
    }catch(e){console.error('pollOAuth fetch failed:',e)}
  }
  document.getElementById('oauthStatusText').textContent='已超时（仍可在下方粘贴回调 URL）';
  if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
}
function copyOauthUrl(){
  const input=document.getElementById('oauthUrl');
  const text=input.value;
  if(!text){toast('URL 为空','warn');return}
  const showSuccess=()=>toast('URL 已复制','success');
  const showFailed=(err)=>{console.error('copyOauthUrl failed:',err);toast('复制失败，请手动选中复制','error')};
  /* Prefer Clipboard API when available (HTTPS / localhost) */
  if(navigator.clipboard&&typeof navigator.clipboard.writeText==='function'){
    navigator.clipboard.writeText(text).then(showSuccess).catch(err=>{
      /* Permission denied or not allowed — fall back to execCommand */
      if(fallbackCopy(input))showSuccess();else showFailed(err);
    });
    return;
  }
  /* Non-secure context (HTTP LAN IP) — use legacy execCommand path */
  if(fallbackCopy(input))showSuccess();else showFailed(new Error('Clipboard API unavailable'));
  function fallbackCopy(el){
    try{
      el.focus();el.select();
      /* For iOS / mobile */
      if(el.setSelectionRange&&el.value.length)el.setSelectionRange(0,el.value.length);
      const ok=document.execCommand('copy');
      el.blur();
      return ok;
    }catch(e){return false}
  }
}

// --- Logs ---
let logEventSource=null;
function connectLogStream(){
  if(logEventSource){try{logEventSource.close()}catch{}}
  const sseUrl=API+'/admin/api/logs/stream?token='+encodeURIComponent(authToken);
  try{
    logEventSource=new EventSource(sseUrl);
    logEventSource.onmessage=e=>{
      try{
        const d=JSON.parse(e.data);
        appendLog(d);
      }catch(err){console.error('log SSE parse failed:',err,e.data)}
    };
    logEventSource.onerror=()=>{
      try{logEventSource.close()}catch(err){void err}
      setTimeout(()=>{if(authToken)connectLogStream()},3000);
    };
  }catch(err){console.error('connectLogStream failed:',err)}
}
function appendLog(entry){
  logLines.push(entry);
  if(logLines.length>2000)logLines=logLines.slice(-1000);
  renderLogs();
}
function renderLogs(){
  const box=document.getElementById('logBox');
  const level=document.getElementById('logLevel').value;
  const search=document.getElementById('logSearch').value.toLowerCase();
  const filtered=logLines.filter(l=>{
    if(level&&l.level!==level)return false;
    if(search&&!l.message.toLowerCase().includes(search))return false;
    return true;
  });
  /* Render with colored levels */
  box.innerHTML=filtered.map(l=>{
    const lvl=l.level||'info';
    return `<div class="log-line"><span class="log-time">[${l.time}]</span> <span class="log-level-${lvl}">[${lvl}]</span> ${escapeHtml(l.message)}</div>`;
  }).join('');
  if(document.getElementById('logAutoScroll').checked)box.scrollTop=box.scrollHeight;
}
function clearLogs(){logLines=[];document.getElementById('logBox').innerHTML=''}
document.getElementById('logLevel').addEventListener('change',renderLogs);
document.getElementById('logSearch').addEventListener('input',renderLogs);

    <td><input value="${escapeHtml(r.pattern)}" onchange="routingRules[${i}].pattern=this.value" placeholder="例如 glm-5*"></td>
    <td><select onchange="routingRules[${i}].provider=this.value"><option value="zai"${r.provider==='zai'?' selected':''}>Z.AI</option><option value="bigmodel"${r.provider==='bigmodel'?' selected':''}>Bigmodel</option></select></td>
    <td><input value="${escapeHtml(r.endpoint||'')}" onchange="routingRules[${i}].endpoint=this.value" placeholder="默认"></td>
    <td><input value="${escapeHtml(r.note||'')}" onchange="routingRules[${i}].note=this.value" placeholder="可选"></td>
    <td class="td-actions"><button class="btn btn-danger btn-sm" onclick="routingRules.splice(${i},1);renderRoutingRules()">删除</button></td>
  </tr>`).join('');
}
async function loadRoutingRules(){
  try{
    const r=await fetch(API+'/admin/api/routing-rules',{headers:authHeaders()});
    const d=await r.json();
    routingRules=(d.rules||[]).map(r=>({pattern:r.pattern,provider:r.provider,endpoint:r.endpoint||'',note:r.note||''}));
    renderRoutingRules();
  }catch(e){console.error('loadRoutingRules failed:',e);toast('加载路由规则失败','error')}
}
async function saveRoutingRules(){
  for(let i=0;i<routingRules.length;i++){
    const r=routingRules[i];
    if(!r.pattern||!r.pattern.trim()){toast(`第 ${i+1} 条规则缺少模型匹配规则`,'error');return}
    if(r.provider!=='zai'&&r.provider!=='bigmodel'){toast(`第 ${i+1} 条规则的提供商无效`,'error');return}
  }
  try{
    const r=await fetch(API+'/admin/api/routing-rules',{method:'PUT',headers:authHeaders(),body:JSON.stringify({rules:routingRules})});
    const d=await r.json();
    if(r.ok){
      routingRules=(d.rules||[]).map(r=>({pattern:r.pattern,provider:r.provider,endpoint:r.endpoint||'',note:r.note||''}));
      renderRoutingRules();
      toast(`已保存 ${routingRules.length} 条路由规则`,'success');
    }else{
      toast(d.error||'保存失败','error');
    }
  }catch(e){console.error('saveRoutingRules failed:',e);toast('保存失败','error')}
}

async function loadStats(){
  try{
    const r=await fetch(API+'/admin/api/stats',{headers:authHeaders()});
    const d=await r.json();
    statsData=d;
    document.getElementById('statTotal').textContent=d.total||0;
    document.getElementById('statSuccess').textContent=d.success||0;
    document.getElementById('statFailed').textContent=d.failed||0;
    document.getElementById('statRetried').textContent=d.retried||0;
    document.getElementById('statRequests').textContent=d.total||0;
    document.getElementById('statRetries').textContent=d.retried||0;
    /* Update progress bar */
    const total=d.total||0;
    const successPct=total?(d.success/total*100):0;
    const failedPct=total?(d.failed/total*100):0;
    const retriedPct=total?(d.retried/total*100):0;
    const progress=document.getElementById('statsProgress');
    if(progress){
      progress.children[0].style.width=successPct+'%';
      progress.children[1].style.width=failedPct+'%';
      progress.children[2].style.width=retriedPct+'%';
    }
    const tb=document.getElementById('recentRequests');
    tb.innerHTML=(d.requests||[]).slice(-20).reverse().map(r=>`<tr>
      <td class="td-mute">${r.id}</td><td class="td-mute">${r.time}</td><td class="td-mono">${r.model}</td>
      <td><span class="badge ${r.status<300?'badge-success':r.status<500?'badge-warn':'badge-danger'}">${r.status}</span></td>
      <td class="td-mono">${r.ttfb||'-'}</td><td class="td-mono">${r.tokens||'-'}</td>
    </tr>`).join('')||'<tr><td colspan="6"><div class="empty-state"><div class="empty-text">暂无请求</div></div></td></tr>';
    const mu=document.getElementById('modelUsage');
    mu.innerHTML=Object.entries(d.models||{}).map(([m,v])=>`<tr><td class="td-mono">${m}</td><td>${v.count}</td><td class="td-mono">${v.avgTtfb}ms</td><td class="td-mono">${v.tokens}</td></tr>`).join('')
      ||'<tr><td colspan="4"><div class="empty-state"><div class="empty-text">暂无数据</div></div></td></tr>';
  }catch(e){console.error('loadStats failed:',e)}
}
setInterval(()=>{if(authToken)loadStats()},10000);

async function resetStats(){
  if(!confirm('确认重置所有统计数据？此操作不可撤销。'))return;
  try{
    const r=await fetch(API+'/admin/api/stats',{method:'DELETE',headers:authHeaders()});
    if(r.ok){toast('统计已重置','success');loadStats()}
    else toast('重置失败','error');
  }catch(e){console.error('resetStats failed:',e);toast('重置失败','error')}
}

function toast(msg,type='success'){
  const container=document.getElementById('toastContainer');
  if(!container)return;
  const el=document.createElement('div');
  el.className='toast toast-'+type;
  const icon=TOAST_ICONS[type]||TOAST_ICONS.info;
  el.innerHTML=`<span class="toast-icon">${icon}</span><span class="toast-msg">${escapeHtml(msg)}</span>`;
  el.addEventListener('click',()=>{
    el.classList.add('leaving');
    setTimeout(()=>el.remove(),200);
  });
  container.appendChild(el);
  setTimeout(()=>{
    if(el.parentNode){
      el.classList.add('leaving');
      setTimeout(()=>el.remove(),200);
    }
  },3000);
}
