async function loadUptimeFromServer(){
  // Guard: skip the request when not authenticated (e.g. after logout).
  // Without this, the interval keeps firing 401s forever.
  if(!authToken)return;
  try{
    const r=await fetch(API+'/admin/api/stats',{headers:authHeaders()});
    const d=await r.json();
    const s=Math.floor((d.uptime||0)/1000);
    const h=Math.floor(s/3600),m=Math.floor(s%3600/60),ss=s%60;
    const uptimeStr=`${h}h ${String(m).padStart(2,'0')}m ${String(ss).padStart(2,'0')}s`;
    document.getElementById('statUptime').textContent=uptimeStr;
    document.getElementById('heroUptime').textContent=uptimeStr;
    // Also refresh overview stats (total requests, retries) so the overview
    // page doesn't sit stale while the user is on it.
    document.getElementById('statRequests').textContent=d.total||0;
    document.getElementById('statRetries').textContent=d.retried||0;
    // vceshi0.0.7+: also update _cachedByCredential so the accounts page
    // usage counts stay fresh without waiting for the 10s loadStats interval.
    // Only re-render the accounts table if it's currently visible.
    if(d.byCredential){
      _cachedByCredential=d.byCredential;
      const accountsPage=document.getElementById('page-accounts');
      // Use classList check — pages use .active class to show/hide.
      if(accountsPage&&accountsPage.classList.contains('active')&&_cachedAccounts.length>0){
        filterAccounts();
      }
    }
  }catch{}
}

async function saveSettings(){
  // --- 客户端预校验重试字段（服务端 validateConfigForSave 严格校验，
  // 不符合的值会导致整次保存失败 + 真实错误被旧 toast 吞掉，表现为
  // 「一改重试，所有 tab 都保存失败」。这里提前拦截，给用户明确提示）。
  const maxRetries=parseInt(document.getElementById('cfgRetryMax').value,10);
  const initialDelayMs=parseInt(document.getElementById('cfgRetryDelay').value,10);
  const maxDelayMs=parseInt(document.getElementById('cfgRetryMaxDelay').value,10);
  const backoffFactor=parseFloat(document.getElementById('cfgRetryBackoff').value);
  const statusesRaw=document.getElementById('cfgRetryStatuses').value;
  const credSwitch=parseInt(document.getElementById('cfgRetryCredSwitch').value,10);
  const emptyStreamSwitch=parseInt(document.getElementById('cfgRetryEmptyStreamSwitch').value,10);
  // 与服务端 validateConfigForSave 一致：100-599
  const statuses=statusesRaw.split(',').map(s=>s.trim()).filter(Boolean)
    .map(s=>parseInt(s,10))
    .filter(n=>Number.isFinite(n)&&n>=100&&n<=599);
  if(Number.isFinite(maxRetries)&&maxRetries<0){
    toast('最大重试次数必须 >= 0（无上限）','error');return;
  }
  if(Number.isFinite(initialDelayMs)&&(initialDelayMs<0||initialDelayMs>60000)){
    toast('初始延迟必须在 0-60000ms 之间','error');return;
  }
  if(Number.isFinite(maxDelayMs)&&(maxDelayMs<0||maxDelayMs>300000)){
    toast('最大延迟必须在 0-300000ms 之间','error');return;
  }
  if(Number.isFinite(backoffFactor)&&backoffFactor<=0){
    toast('退避因子必须大于 0','error');return;
  }
  // 检查用户输入的状态码格式：如果输入了非空但全部解析失败，提示
  if(statusesRaw.trim()&&statuses.length===0){
    toast('可重试状态码必须是 100-599 之间的数字（逗号分隔）','error');return;
  }
  // 凭证切换阈值：非负整数，0 = 禁用
  if(Number.isFinite(credSwitch)&&credSwitch<0){
    toast('凭证切换阈值必须 >= 0（0 = 禁用）','error');return;
  }
  // 空回切换阈值：非负整数，0 = 禁用（回退到普通阈值）
  if(Number.isFinite(emptyStreamSwitch)&&emptyStreamSwitch<0){
    toast('空回切换阈值必须 >= 0（0 = 禁用，回退到普通凭证切换阈值）','error');return;
  }

  const cfg={
    server:{port:+document.getElementById('cfgPort').value,host:document.getElementById('cfgHost').value},
    provider:document.getElementById('cfgProvider').value,
    plan:document.getElementById('cfgPlan').value,
    auth:{mode:document.getElementById('cfgAuthMode').value},
    defaultModel:document.getElementById('cfgDefaultModel').value,
    models:document.getElementById('cfgModels').value.split('\n').map(s=>s.trim()).filter(Boolean),
    retry:{
      // 空值兜底为合理默认，避免 +"" === 0 把字段变成 0；
      // 服务端 loadConfig 的 resolvePositive* 也有兜底，这里保持一致。
      maxRetries:Number.isFinite(maxRetries)?maxRetries:3,
      initialDelayMs:Number.isFinite(initialDelayMs)?initialDelayMs:1000,
      maxDelayMs:Number.isFinite(maxDelayMs)?maxDelayMs:8000,
      backoffFactor:Number.isFinite(backoffFactor)&&backoffFactor>0?backoffFactor:2,
      retryableStatuses:statuses.length>0?statuses:[529],
      credentialSwitchThreshold:Number.isFinite(credSwitch)&&credSwitch>=0?credSwitch:5,
      emptyStreamSwitchThreshold:Number.isFinite(emptyStreamSwitch)&&emptyStreamSwitch>=0?emptyStreamSwitch:3,
    },
    identity:{appVersion:document.getElementById('cfgAppVer').value,sourceTitle:document.getElementById('cfgSrcTitle').value,refererOrigin:document.getElementById('cfgReferer').value},
    logging:{level:document.getElementById('cfgLogLevel').value,verbose:document.getElementById('cfgVerboseLog').checked,debug:document.getElementById('cfgDebugLog').checked},
    forceStreamAnthropic:document.getElementById('cfgForceStreamAnthropic').checked,
    thinkingLevel:document.getElementById('cfgThinkingLevel').value==='high'?'high':'max'
  };
  const apiKey=document.getElementById('cfgApiKey').value;
  const proxyApiKey=document.getElementById('cfgProxyApiKey').value;
  if(apiKey&&apiKey!=='***configured***')cfg.auth.apiKey=apiKey;
  if(proxyApiKey&&proxyApiKey!=='***configured***')cfg.auth.proxyApiKey=proxyApiKey;
  try{
    const r=await fetch(API+'/admin/api/config',{method:'PUT',headers:authHeaders(),body:JSON.stringify(cfg)});
    if(r.ok){
      const d=await r.json().catch(()=>({}));
      if(d.requiresRestart){
        toast(`设置已保存，但以下字段需重启生效：${(d.restartFields||[]).join(', ')}`,'warn');
      }else{
        toast('设置已保存（已热生效）','success');
      }
    }
    else{
      const d=await r.json().catch(()=>({}));
      // 服务端错误结构为 { error: { type, message } }，旧代码误读 d.message，
      // 导致真实错误被吞掉，用户只看到无意义的「保存失败」。
      const msg=d?.error?.message||d?.message||'保存失败';
      toast(msg,'error');
      console.error('saveSettings failed:',{status:r.status,body:d});
    }
  }catch{toast('保存失败','error')}
}

// Pre-fill the ZCode import dropdowns from the server's auto-detection:
// activeProvider selects the provider, and provider×plan combos with no
// detected credential are disabled (greyed out) so the user can't pick a
// combination that would fail. Non-blocking — if detection fails we just
// leave both dropdowns at their default and let the user pick manually.
let _importAvailMap={};
function applyImportAvailability(availMap){
  _importAvailMap=availMap;
  const provSel=document.getElementById('importProvider');
  const planSel=document.getElementById('importPlan');
  if(!provSel||!planSel)return;
  // A provider is importable if either of its plan combos has a credential.
  Array.from(provSel.options).forEach(opt=>{
    opt.disabled=!(availMap[opt.value+'|coding-plan']||availMap[opt.value+'|start-plan']);
  });
  refreshImportPlanOptions();
}
// Re-evaluate which plans are pickable for the currently-selected provider,
// and snap to the first available one if the current choice just got disabled.
function refreshImportPlanOptions(){
  const provSel=document.getElementById('importProvider');
  const planSel=document.getElementById('importPlan');
  if(!provSel||!planSel)return;
  Array.from(planSel.options).forEach(opt=>{
    opt.disabled=!_importAvailMap[provSel.value+'|'+opt.value];
  });
  if(planSel.options[planSel.selectedIndex]&&planSel.options[planSel.selectedIndex].disabled){
    const firstAvail=Array.from(planSel.options).find(o=>!o.disabled);
    if(firstAvail)planSel.value=firstAvail.value;
  }
}
function onImportProviderChange(){refreshImportPlanOptions();}
async function initImportCard(){
  const provSel=document.getElementById('importProvider');
  const planSel=document.getElementById('importPlan');
  if(!provSel||!planSel)return;
  try{
    const r=await fetch(API+'/admin/api/import/detect',{headers:authHeaders()});
    if(!r.ok)return;
    const d=await r.json();
    // available: [{provider, plan, hasApiKey}] — a combo is pickable only if it has a credential.
    const availMap={};
    (d.available||[]).forEach(a=>{availMap[a.provider+'|'+a.plan]=a.hasApiKey;});
    applyImportAvailability(availMap);
    // Pre-fill provider from activeProvider (if that provider is available).
    if(d.activeProvider){
      const opt=Array.from(provSel.options).find(o=>o.value===d.activeProvider);
      if(opt&&!opt.disabled)provSel.value=d.activeProvider;
      refreshImportPlanOptions();
    }
  }catch(e){console.debug('import detect failed:',e)}
}

// Cache of the last-loaded accounts list, used by filterAccounts() to avoid
// re-fetching from the server on every keystroke in the search box.
let _cachedAccounts=[];
let _cachedActiveId=null;

async function loadAccountsList(){
  const tb=document.getElementById('accountsList');
  if(!tb)return;

    const r=await fetch(API+'/admin/api/accounts',{headers:authHeaders()});

/**
 * Re-render the accounts table using the cached list + current filter inputs.
 * Called on every keystroke in the search box and every change of the filter
 * dropdowns — no network call, just a re-filter of _cachedAccounts.
 */
function filterAccounts(){
  const tb=document.getElementById('accountsList');
  if(!tb)return;
  const search=(document.getElementById('accountSearch')?.value||'').trim().toLowerCase();
  const fProvider=document.getElementById('accountFilterProvider')?.value||'';
  const fPlan=document.getElementById('accountFilterPlan')?.value||'';
  let accounts=_cachedAccounts;
  // Apply filters
  if(fProvider)accounts=accounts.filter(a=>a.provider===fProvider);
  if(fPlan)accounts=accounts.filter(a=>(a.plan||'coding-plan')===fPlan);
  if(search){
    accounts=accounts.filter(a=>{
      const label=(a.label||'').toLowerCase();
      const key=(a.apiKeyMask||'').toLowerCase();
      return label.includes(search)||key.includes(search);
    });
  }
  const activeId=_cachedActiveId;
  if(!_cachedAccounts.length){
    tb.innerHTML='<tr><td colspan="9"><div class="empty-state"><div class="empty-icon"><svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M17 21v-2a4 4 0 0 0-4-4H5a4 4 0 0 0-4 4v2"/><circle cx="9" cy="7" r="4"/></svg></div><div class="empty-text">暂无存储的账号</div><div class="empty-hint">请使用 OAuth 或"添加 API Key"创建</div></div></td></tr>';

    return;
  }
  tb.innerHTML=accounts.map(a=>{
    const isActive=a.id===activeId;
    const isDisabled=!!a.disabled;
    const created=new Date(a.createdAt).toLocaleString();
    const planVal=a.plan||'coding-plan';
    const hasProxy=!!(a.proxy&&a.proxy.trim());

    // vceshi0.0.6+: status column shows both active + disabled badges
    let statusBadge;
    if(isActive) statusBadge='<span class="badge badge-success"><span class="badge-dot"></span>已激活</span>';
    else statusBadge='<span class="badge badge-muted">未激活</span>';

/** Open a read-only detail modal showing all credential info. */
async function openAccountDetail(id){
  // Render the modal shell immediately with a loading state so the click
  // feels responsive — previously the user clicked and saw nothing for
  // 100-500ms while the fetch was in flight, which read as "lag".
  const loadingHtml=`<div class="modal-overlay" onclick="if(event.target===this)closeModal('accountDetailModal')">
    <div class="modal" style="max-width:640px">
      <div class="modal-header">
        <h3>账号详情</h3>
        <button class="modal-close" onclick="closeModal('accountDetailModal')">×</button>
      </div>
      <div class="modal-body">
        <div class="empty-state" style="padding:var(--s-7) var(--s-4)">
          <div class="empty-icon" style="animation:shimmer 1.4s infinite"><svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><circle cx="11" cy="11" r="8"/><path d="M21 21l-4.35-4.35"/></svg></div>
          <div class="empty-text">加载凭证信息...</div>
        </div>
      </div>
      <div class="modal-footer">
        <button class="btn btn-primary" onclick="closeModal('accountDetailModal')">关闭</button>
      </div>
    </div>
  </div>`;
  showModal('accountDetailModal',loadingHtml);
  try{
    const r=await fetch(API+'/admin/api/accounts/export-single?id='+encodeURIComponent(id),{headers:authHeaders()});
    if(!r.ok){
      const d=await r.json().catch(()=>({}));
      closeModal('accountDetailModal');
      toast(d.error||'查看失败','error');return;
    }
    const d=await r.json();
    const acc=d.account;
    if(!acc){closeModal('accountDetailModal');toast('账号不存在','error');return;}
    const cred=acc.credential||{};
    const created=new Date(acc.createdAt).toLocaleString();
    // Build a structured key-value table for display. Sensitive fields
    // (apiKey/secret/jwt) are masked by default with a "show" toggle.
    const rows=[
      ['名称',cred.name||acc.label||'(无)'],
      ['自动标签',acc.label],
      ['邮箱',cred.email||'(无)'],
      ['提供商',cred.provider],
      ['套餐',cred.plan||'(未设置)'],
      ['创建时间',created],
      ['账号 ID',acc.id],
      ['API Key',cred.apiKey?{secret:cred.apiKey,mask:maskForDisplay(cred.apiKey)}:'(无)'],
      ['Secret',cred.secret?{secret:cred.secret,mask:maskForDisplay(cred.secret)}:'(无)'],
      ['JWT',cred.jwt?{secret:cred.jwt,mask:cred.jwt.substring(0,12)+'...'}:'(无)'],
      ['User ID',cred.userId||'(无)'],
      ['代理',cred.proxy||'(无)'],
    ];
    const rowsHtml=rows.map(([k,v])=>{
      if(v&&typeof v==='object'&&v.secret){
        // Sensitive field with show/hide toggle.
        // vceshi0.0.7+: use data-secret attribute + event delegation instead
        // of inline onclick with the secret embedded. The old code did
        // `onclick="toggleSecret('${fieldId}','${escapeHtml(v.secret).replace(/'/g,"\\'")}')"`,
        // which broke when the secret contained a single quote: escapeHtml
        // converts ' to &#39;, the HTML parser decodes it back to ', and the
        // JS engine sees `toggleSecret('id','...with ' in it...')` — broken JS.
        // Storing the secret in a data attribute and reading it via DOM
        // avoids the JS-context escaping problem entirely.
        const fieldId='secret-'+k.replace(/[^a-zA-Z0-9]/g,'')+'-'+Date.now()+'-'+Math.random().toString(36).slice(2,6);
        const secretAttr=escapeHtml(JSON.stringify(v.secret));
        return `<tr><td class="td-mute">${k}</td><td><span class="td-mono" id="${fieldId}">${escapeHtml(v.mask)}</span> <button class="btn btn-ghost btn-sm" data-toggle-secret="${fieldId}" data-secret="${secretAttr}">显示</button></td></tr>`;
      }

/** Mask a sensitive string for display: show first 8 + last 4 chars. */
function maskForDisplay(s){
  if(!s)return '';
  if(s.length<=12)return s;
  return s.substring(0,8)+'...'+s.slice(-4);
}

/** Toggle a secret field between masked and plaintext. */
function toggleSecret(fieldId,fullValue){
  const el=document.getElementById(fieldId);
  if(!el)return;
  if(el.dataset.revealed==='1'){
    el.textContent=maskForDisplay(fullValue);
    el.dataset.revealed='0';
    el.nextElementSibling.textContent='显示';
  }else{
    el.textContent=fullValue;
    el.dataset.revealed='1';
    el.nextElementSibling.textContent='隐藏';
  }
}

/** Export a single account's full credential JSON as a downloadable file. */
async function exportSingleAccountJson(id){
  try{
    const r=await fetch(API+'/admin/api/accounts/export-single?id='+encodeURIComponent(id),{headers:authHeaders()});
    if(!r.ok){
      const d=await r.json().catch(()=>({}));

    }

    const acc=d.account;
    if(!acc){toast('账号不存在','error');return;}

    const exportObj={

      name:acc.credential?.name||null,
      email:acc.credential?.email||null,
      createdAt:acc.createdAt,
      credential:acc.credential,
    };
    const jsonStr=JSON.stringify(exportObj,null,2);

    const blob=new Blob([jsonStr],{type:'application/json'});

    toast('已导出 JSON 文件','success');

// Per-account proxy modal (v2.1.4.1test6+)
// ============================================================
// Replaces the inline proxy input with a full modal dialog. The modal lets
// users pick proxy type (none/http/https/socks5/socks5h), enter host/port/
// user/pass separately, test connectivity, and save. Internally we still
// store a single URL string on Credential.proxy — the modal just parses
// and assembles that URL for the user.

/** Currently focused account id in the proxy modal. */
let proxyModalAccountId=null;
let proxyModalProvider=null;

/**
 * Build a proxy URL string from individual modal fields.
 * Returns "" when type is "none" (meaning "direct connection, no proxy").
 * Returns the assembled URL otherwise.
 */
function buildProxyUrlFromModal(){
  const type=document.getElementById('proxyType').value;
  if(type==='none')return '';
  const host=document.getElementById('proxyHost').value.trim();
  const port=document.getElementById('proxyPort').value.trim();
  const user=document.getElementById('proxyUser').value.trim();
  const pass=document.getElementById('proxyPass').value;
  if(!host){
    // Caller (test/save) is expected to validate; return empty so the
    // validation message is consistent ("host is required").
    return '';
  }
  let auth='';
  if(user||pass){
    // URL-encode user/pass to handle special chars (@, :, /, etc.)
    auth=encodeURIComponent(user)+(pass?(':'+encodeURIComponent(pass)):'')+'@';
  }
  // Default port hints: http(s)→80/443 typically omitted; socks5→1080.
  // We don't auto-fill — let the user specify. Empty port is allowed.
  const portPart=port?(':'+port):'';
  return `${type}://${auth}${host}${portPart}`;
}

/**
 * Parse a stored proxy URL back into {type, host, port, user, pass}.
 * Used when opening the modal to populate fields with the current value.
 * Returns {type:'none'} for empty/invalid URLs.
 */
function parseProxyUrl(url){
  const empty={type:'none',host:'',port:'',user:'',pass:''};
  if(!url||!url.trim())return empty;
  const trimmed=url.trim();
  const m=/^(https?|socks5h?):\/\/(?:([^:@/]+)(?::([^@/]*))?@)?([^:/]+)(?::(\d+))?\/?$/i.exec(trimmed);
  if(!m)return empty; // unrecognized format → reset to none
  return {
    type:m[1].toLowerCase(),
    user:decodeURIComponent(m[2]||''),
    pass:decodeURIComponent(m[3]||''),
    host:m[4]||'',
    port:m[5]||'',
  };
}

/** Open the proxy modal for a given account. */
function openProxyModal(id,currentProxy,provider){
  proxyModalAccountId=id;
  proxyModalProvider=(provider==='bigmodel')?'bigmodel':'zai';
  const parsed=parseProxyUrl(currentProxy);
  document.getElementById('proxyType').value=parsed.type;
  document.getElementById('proxyHost').value=parsed.host;
  document.getElementById('proxyPort').value=parsed.port;
  document.getElementById('proxyUser').value=parsed.user;
  document.getElementById('proxyPass').value=parsed.pass;

}

/** Format a unix-seconds timestamp (or ISO string) for display. */
function formatQuotaTime(v){
  if(v===null||v===undefined||v==='')return null;
  const n=typeof v==='number'?v:Date.parse(v);
  if(typeof n==='number'&&Number.isFinite(n)){
    // Treat plain integers as unix seconds (billing/ends_at), ms otherwise
    const ms=n>1e12?n:n*1000;
    const dt=new Date(ms);
    const s=dt.toLocaleString();
    return Number.isFinite(dt.getTime())?s:null;
  }
  return null;
}

/** Render a QuotaResult into the modal body. */
function renderQuota(d){
  const body=document.getElementById('quotaBody');
  const reason=d.unavailableReason;
  const reasonMap={'no_plan':'该凭证没有有效的套餐（可能未开通 coding/start plan）','unavailable':'上游暂时无法返回额度信息（可能凭证已过期或上游限流）','not_configured':'凭证未配置，无法查询'};
  if(reason){
    body.innerHTML='<div style="padding:20px;color:var(--warning,var(--text-2));font-size:13px;line-height:1.6;">'+escapeHtml(reasonMap[reason]||'无法查询额度')+'</div>';
    return;
  }
  const planLine=d.planName?('<div style="color:var(--text-2);font-size:12px;margin-bottom:4px;">套餐：<b style="color:var(--text);">'+escapeHtml(d.planName)+'</b></div>'):'';
  const exp=formatQuotaTime(d.expireTime);
  const expLine=exp?('<div style="color:var(--text-3);font-size:12px;margin-bottom:14px;">到期：'+escapeHtml(exp)+'</div>'):'';
  // Header row: plan + provider
  const header='<div style="font-size:11px;color:var(--text-3);margin-bottom:6px;text-transform:uppercase;letter-spacing:.04em;">'+escapeHtml(d.plan)+' · '+escapeHtml(d.provider)+'</div>';
  // Build the main body from per-limit entries. Each limit gets its own
  // progress bar — important because start-plan grants are per-MODEL (e.g.
  // GLM-5.2 exhausted while GLM-5-Turbo is full), and summing them into one
  // percentage would mislead ("40%" when one model is at 0%).
  const limits=(Array.isArray(d.limits)?d.limits:[]).filter(l=>l.remaining!=null&&l.total!=null);
  let limitsBlock='';
  if(limits.length){
    limitsBlock='<div style="display:flex;flex-direction:column;gap:12px;">'+
      limits.map(l=>{
        const pct=l.total>0?Math.round(l.remaining/l.total*1000)/10:0;
        const color=pct<20?'var(--danger)':(pct<50?'var(--warning,#e0a800)':'var(--success,#22c55e)');
        const next=formatQuotaTime(l.nextResetTime);
        return '<div>'+
          '<div style="display:flex;justify-content:space-between;align-items:baseline;margin-bottom:4px;">'+
            '<span style="font-size:13px;color:var(--text);font-weight:500;">'+escapeHtml(l.label||l.type||'-')+'</span>'+
            '<span style="font-size:12px;color:var(--text-2);"><b style="color:'+color+';font-size:14px;">'+pct+'%</b> 剩余 '+escapeHtml(fmtNum(l.remaining))+' / '+escapeHtml(fmtNum(l.total))+'</span>'+
          '</div>'+
          '<div style="height:8px;background:var(--bg-elev,var(--border));border-radius:4px;overflow:hidden;">'+
            '<div style="height:100%;width:'+Math.max(0,Math.min(100,pct))+'%;background:'+color+';transition:width .3s;"></div>'+
          '</div>'+
          (next?'<div style="font-size:11px;color:var(--text-3);margin-top:3px;">每日重置 · '+escapeHtml(next)+'</div>':'')+
        '</div>';
      }).join('')+
    '</div>';
  }else{
    // No per-limit data (e.g. coding-plan returned a level but no limits, or
    // balance had no totals). Fall back to the aggregate remaining if present.
    const rem=d.remaining;
    if(rem&&rem.total>0){
      const pct=Math.round(rem.percentage);
      const color=pct<20?'var(--danger)':(pct<50?'var(--warning,#e0a800)':'var(--success,#22c55e)');
      limitsBlock='<div style="font-size:14px;color:var(--text-2);">剩余 <b style="color:'+color+';font-size:18px;">'+pct+'%</b>（'+escapeHtml(fmtNum(rem.count))+' / '+escapeHtml(fmtNum(rem.total))+'）</div>';
    }else{
      limitsBlock='<div style="color:var(--text-3);font-size:12px;">无可用额度数据</div>';
    }
  }
  body.innerHTML=header+planLine+expLine+limitsBlock;
}

async function importAccountsFromFile(input){
  if(!input.files||!input.files[0])return;
  const file=input.files[0];
  try{
    const text=await file.text();
    const data=JSON.parse(text);
    // Accept multiple export formats so users can re-import files produced by
    // any of the dashboard's export buttons (single-account JSON, multi-account
    // backup, or full v2 store envelope). Previously only `{accounts:[...]}` was
    // accepted, which silently rejected the per-row "导出" button's output
    // (shape: `{id,label,name,email,createdAt,credential}`) with a misleading
    // "缺少 accounts 数组" error.
    let accounts=null;
    if(Array.isArray(data)){
      // Bare array of account objects.
      accounts=data;
    }else if(data&&Array.isArray(data.accounts)){
      // Multi-account backup: {accounts:[...], exportedAt, version} or full
      // v2 store envelope: {version:2, activeId, accounts:[...]}.
      accounts=data.accounts;
    }else if(data&&typeof data.id==='string'&&data.credential&&typeof data.credential.apiKey==='string'){
      // Single-account export from the per-row "导出" button.
      accounts=[data];
    }
    if(!accounts||!Array.isArray(accounts)||accounts.length===0){
      toast('文件无效：未识别到可导入的账号数据（支持单账号导出、多账号备份、v2 store 三种格式）','error');
      return;
    }
    if(!confirm(`从"${file.name}"导入 ${accounts.length} 个账号？\nID 匹配的现有账号将被更新。`))return;
    const r=await fetch(API+'/admin/api/accounts/import',{method:'POST',headers:authHeaders(),body:JSON.stringify({accounts})});
    const d=await r.json().catch(()=>({}));
    if(r.ok){toast(`导入完成：新增 ${d.added}，更新 ${d.updated}`,'success');loadAccountInfo();loadAccountsList();loadCredStore()}
    else toast(d.error?.message||d.message||'导入失败','error');
  }catch(e){console.error('importAccounts failed:',e);toast('导入失败：文件无效','error')}
  input.value='';
}
function escapeHtml(s){return String(s).replace(/[&<>"']/g,c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]))}

let oauthPollCancelled=false;

async function startOAuth(){
  const provider=document.getElementById('oauthProvider').value;
  const plan=document.getElementById('oauthPlan').value;
  document.getElementById('oauthFlowStatus').style.display='block';
  document.getElementById('oauthDot').className='oauth-dot off';
  document.getElementById('oauthStatusText').textContent='初始化中...';
  document.getElementById('oauthUrlBox').style.display='none';
  document.getElementById('callbackResult').style.display='none';
  try{
    const r=await fetch(API+'/admin/api/oauth/init',{method:'POST',headers:authHeaders(),body:JSON.stringify({provider,plan})});
    const d=await r.json();
    if(!r.ok)throw new Error(d.error||'Init failed');
    // vceshi0.0.7+: cancel any previous poll loop before starting a new one.
    // Without this, starting a new OAuth flow while the old one was still
    // polling would leave both loops running — the old loop could update
    // the UI with stale "ready" status after the user moved on.
    if(oauthPollCancelled===false){
      oauthPollCancelled=true; // signal the previous loop to stop
    }
    currentFlowId=d.flowId;
    sessionStorage.setItem('zcodeOAuthFlowId',d.flowId);
    document.getElementById('oauthDot').className='oauth-dot on';
    document.getElementById('oauthStatusText').textContent='等待授权中...';
    document.getElementById('oauthUrlBox').style.display='block';
    document.getElementById('oauthUrl').value=d.authorizeUrl;
    oauthExpiresAt=d.expiresAt||0;
    startOauthCountdown();
    // Reset the cancellation flag for the new poll loop, then start it.
    // The flag is captured by closure so the loop can check it on each iteration.
    oauthPollCancelled=false;
    pollOAuth(d.flowId);
  }catch(e){toast(e.message,'error');document.getElementById('oauthStatusText').textContent='失败：'+e.message}
}
function startOauthCountdown(){
  if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
  const update=()=>{
    const remainingMs=oauthExpiresAt?oauthExpiresAt-Date.now():0;
    if(remainingMs<=0){
      document.getElementById('oauthCountdown').textContent='已过期';
      clearInterval(oauthExpiryTimer);
      return;
    }
    const s=Math.floor(remainingMs/1000);
    const m=Math.floor(s/60),ss=s%60;
    document.getElementById('oauthCountdown').textContent=`${m}:${String(ss).padStart(2,'0')}`;
  };
  update();
  oauthExpiryTimer=setInterval(update,1000);
}
async function pollOAuth(flowId){
  // vceshi0.0.7+: capture the cancellation flag at loop start. If a new
  // flow starts later, it will set oauthPollCancelled=true; we check the
  // shared flag on each iteration and exit if it's been flipped.
  for(let i=0;i<120;i++){
    // Bail out if a new OAuth flow was started (or callback submitted).
    if(oauthPollCancelled){
      // Don't update UI — the new flow's pollOAuth will handle it.
      return;
    }
    await new Promise(r=>setTimeout(r,2000));
    if(oauthPollCancelled)return; // re-check after the 2s sleep
    try{
      const r=await fetch(API+'/admin/api/oauth/poll?flowId='+flowId,{headers:authHeaders()});
      const d=await r.json();
      if(d.status==='ready'){
        document.getElementById('oauthDot').className='oauth-dot on';
        document.getElementById('oauthStatusText').innerHTML='<span class="badge badge-success"><span class="badge-dot"></span>已授权！</span>';
        document.getElementById('oauthUrlBox').style.display='none';
        if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
        toast('OAuth 登录成功','success');
        currentFlowId=null;
        sessionStorage.removeItem('zcodeOAuthFlowId');
        // vceshi0.0.7+: refresh the accounts list too — the new account
        // won't appear in the multi-account table without this.
        loadAccountInfo();loadCredStore();loadAccountsList();
        return;
      }
      if(d.status==='failed'){
        document.getElementById('oauthDot').className='oauth-dot off';
        document.getElementById('oauthStatusText').textContent='授权失败';
        if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
        // vceshi0.0.5+: surface the error message if present (previously
        // the dashboard couldn't tell the user WHY the flow failed).
        if(d.error){
          toast('OAuth 失败：'+d.error,'error');
        }else{
          toast('OAuth 失败','error');
        }
        return;
      }
      // vceshi0.0.7+: handle 'expired' status — backend returns this when
      // the flow's expiresAt has passed. Previously the dashboard would
      // keep polling for 4 minutes, showing nothing to the user.
      if(d.status==='expired'){
        document.getElementById('oauthDot').className='oauth-dot off';
        document.getElementById('oauthStatusText').textContent='授权已过期，请重新开始登录';
        if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
        toast('OAuth 授权已过期，请重新开始','warn');
        currentFlowId=null;
        sessionStorage.removeItem('zcodeOAuthFlowId');
        return;
      }
    }catch(e){console.error('pollOAuth fetch failed:',e)}
  }
  document.getElementById('oauthStatusText').textContent='已超时（仍可在下方粘贴回调 URL）';
  if(oauthExpiryTimer)clearInterval(oauthExpiryTimer);
}
function copyOauthUrl(){
  const input=document.getElementById('oauthUrl');
  const text=input.value;
  if(!text){toast('URL 为空','warn');return}
  const showSuccess=()=>toast('URL 已复制','success');
  const showFailed=(err)=>{console.error('copyOauthUrl failed:',err);toast('复制失败，请手动选中复制','error')};
  /* Prefer Clipboard API when available (HTTPS / localhost) */
  if(navigator.clipboard&&typeof navigator.clipboard.writeText==='function'){
    navigator.clipboard.writeText(text).then(showSuccess).catch(err=>{
      /* Permission denied or not allowed — fall back to execCommand */
      if(fallbackCopy(input))showSuccess();else showFailed(err);
    });
    return;
  }
  /* Non-secure context (HTTP LAN IP) — use legacy execCommand path */
  if(fallbackCopy(input))showSuccess();else showFailed(new Error('Clipboard API unavailable'));
  function fallbackCopy(el){
    try{
      el.focus();el.select();
      /* For iOS / mobile */
      if(el.setSelectionRange&&el.value.length)el.setSelectionRange(0,el.value.length);
      const ok=document.execCommand('copy');
      el.blur();
      return ok;
    }catch(e){return false}
  }
}

function appendLog(entry){

  if(logLines.length>2000)logLines=logLines.slice(-1000);

/**
 * Determine if a log entry should be renderable as an expandable row.
 *
 * Expandable rows show a ▶ toggle icon and can be clicked to reveal the full
 * message in a styled sub-block. We mark a row as expandable when:
 *   - The message contains [debug] (the new upstream response debug logs)
 *   - The message contains [verbose] (the request headers/body verbose logs)
 *   - The message is longer than 120 chars (any long log benefits from
 *     collapse/expand to keep the panel scannable)
 *
 * When debug mode is OFF, expandable rows are rendered collapsed by default
 * (summary = first 120 chars + …). When ON, the user can click to expand any
 * row, or use the "全部展开" toggle to expand everything at once.
 */
function isLogExpandable(l){
  const msg=l.message||'';
  return msg.includes('[debug]')||msg.includes('[verbose]')||msg.length>120;
}
/** Build the short summary shown when a row is collapsed (max ~120 chars). */
function logSummary(msg){
  if(msg.length<=120)return escapeHtml(msg);
  // Try to break at a natural boundary (space or |) near the 120-char mark
  // so the summary doesn't cut mid-word.
  let cut=120;
  const pipeIdx=msg.indexOf(' | ',80);
  if(pipeIdx>0&&pipeIdx<160)cut=pipeIdx;
  return escapeHtml(msg.slice(0,cut))+' <span style="color:var(--text-3)">…</span>';
}
/** A row is "long" enough to warrant the expand affordance in debug mode. */
const expandedLogSeqs=new Set();
function renderLogs(){
  const box=document.getElementById('logBox');
  const level=document.getElementById('logLevel').value;
  const search=document.getElementById('logSearch').value.toLowerCase();
  const expandAll=document.getElementById('logExpandAll').checked;
  const filtered=logLines.filter(l=>{
    if(level&&l.level!==level)return false;
    if(search&&!l.message.toLowerCase().includes(search))return false;
    return true;
  });
  /* Render with colored levels + expandable rows for debug/verbose/long logs.
   * Each row is a <div class="log-line"> with:
   *   - a toggle icon span (▶/▼/blank)
   *   - the summary (collapsed view, inline)
   *   - a .log-full block (hidden unless expanded) with the complete message
   * Clicking an expandable row toggles its expanded state and re-renders.
   * We track expanded state by seq (monotonic id from the server) so that
   * re-rendering on new log arrival doesn't lose the user's expand selections.
   */
  box.innerHTML=filtered.map(l=>{
    const lvl=l.level||'info';
    const msg=l.message||'';
    const expandable=isLogExpandable(l);
    const expanded=expandAll||expandedLogSeqs.has(l.seq);
    const cls=['log-line',`log-level-${lvl}`];
    if(expandable)cls.push('log-expandable');
    if(expanded)cls.push('log-expanded');
    // Tag badge for debug/verbose rows — makes them visually scannable
    let tag='';
    if(msg.includes('[debug]'))tag='<span class="log-tag-debug">DEBUG</span>';
    else if(msg.includes('[verbose]'))tag='<span class="log-tag-verbose">VERBOSE</span>';
    const toggle='<span class="log-toggle"></span>';
    const summary='<span class="log-summary">'+logSummary(msg)+'</span>';
    // Full content — only rendered for expandable rows to save DOM size.
    // For non-expandable rows the summary IS the full message.
    const full=expandable?'<div class="log-full">'+escapeHtml(msg)+'</div>':'';
    const clickHandler=expandable?`onclick="toggleLogExpand(${l.seq})"`:'';
    return `<div class="${cls.join(' ')}" ${clickHandler}>${toggle}<span class="log-time">[${l.time}]</span> <span class="log-level-${lvl}">[${lvl}]</span>${tag} ${summary}${full}</div>`;
  }).join('');
  if(document.getElementById('logAutoScroll').checked)box.scrollTop=box.scrollHeight;
}
/** Toggle a single log row's expanded state by seq id. */
function toggleLogExpand(seq){
  if(expandedLogSeqs.has(seq)){
    expandedLogSeqs.delete(seq);
  }else{
    expandedLogSeqs.add(seq);
  }
  renderLogs();
}
/**
 * Toggle debug response logging on the server. Sends PUT /admin/api/config
 * with logging.debug flipped. The server hot-swaps the flag, so the next
 * request's [debug] upstream response logs will appear in this panel.
 *
 * Also controls the local UI: when debug is ON, the "全部展开" toggle becomes
 * more useful (debug rows are long and benefit from expansion).
 */
async function toggleLogDebugMode(){
  const checked=document.getElementById('logDebugToggle').checked;
  try{
    // Fetch current config, flip logging.debug, PUT it back.
    const cfgResp=await fetch(API+'/admin/api/config',{headers:authHeaders()});
    const cfg=await cfgResp.json();
    cfg.logging={...(cfg.logging||{}),debug:checked};
    const r=await fetch(API+'/admin/api/config',{method:'PUT',headers:authHeaders(),body:JSON.stringify(cfg)});
    if(r.ok){
      toast(checked?'调试响应日志已开启 — 每个请求将显示上游完整响应':'调试响应日志已关闭','success');
      // When enabling debug, auto-enable "全部展开" so the user immediately
      // sees the full content without having to click each row.
      if(checked){
        document.getElementById('logExpandAll').checked=true;
      }
      renderLogs();
    }else{
      toast('切换调试模式失败','error');
      // Revert the checkbox on failure
      document.getElementById('logDebugToggle').checked=!checked;
    }
  }catch(e){
    console.error('toggleLogDebugMode failed:',e);
    toast('切换调试模式失败','error');
    document.getElementById('logDebugToggle').checked=!checked;
  }
}
/** Sync the log-page debug toggle with the server's current logging.debug state. */
async function syncLogDebugToggle(){
  try{
    const r=await fetch(API+'/admin/api/config',{headers:authHeaders()});
    const c=await r.json();
    document.getElementById('logDebugToggle').checked=!!c.logging?.debug;
  }catch{/* non-fatal — toggle just stays unchecked */}
}

function renderModelMappings(){
  const tb=document.getElementById('modelMappings');
  if(!modelMappings.length){tb.innerHTML='<tr><td colspan="5"><div class="empty-state"><div class="empty-icon"><svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M16 3h5v5"/><path d="M8 3H3v5"/><path d="M21 16v5h-5"/><path d="M3 16v5h5"/><line x1="9" y1="9" x2="15" y2="15"/><line x1="15" y1="9" x2="9" y2="15"/></svg></div><div class="empty-text">暂无模型映射</div><div class="empty-hint">点击右上角"添加映射"创建，例如 gpt-5.5 → glm-5.2</div></div></td></tr>';return}
  // Build datalist from the FULL GLM catalog (provider/models.ts), not just
  // the whitelist — operators may whitelist a subset but still want to
  // map to any catalog model. Also include any whitelisted ids that aren't
  // in the catalog (forward-compat for unreleased models).
  const whitelist=window.__config&&Array.isArray(window.__config.models)?window.__config.models:[];
  const seenIds=new Set();
  const datalistOptions=[];
  for(const m of glmCatalog){
    const k=m.id.toLowerCase();
    if(!seenIds.has(k)){seenIds.add(k);datalistOptions.push(`<option value="${escapeHtml(m.id)}">`);}
  }
  for(const id of whitelist){
    const k=String(id).toLowerCase();
    if(!seenIds.has(k)){seenIds.add(k);datalistOptions.push(`<option value="${escapeHtml(id)}">`);}
  }
  const glmOptions=datalistOptions.join('');
  tb.innerHTML=modelMappings.map((m,i)=>`<tr>
    <td><input value="${escapeHtml(m.from)}" onchange="modelMappings[${i}].from=this.value" placeholder="例如 gpt-5.5" style="font-family:var(--font-mono);font-size:12px"></td>
    <td style="text-align:center;color:var(--text-3)">→</td>
    <td><input value="${escapeHtml(m.to)}" onchange="modelMappings[${i}].to=this.value" placeholder="例如 glm-5.2" list="glm-model-list" style="font-family:var(--font-mono);font-size:12px"></td>
    <td><input value="${escapeHtml(m.note||'')}" onchange="modelMappings[${i}].note=this.value" placeholder="可选"></td>
    <td class="td-actions"><button class="btn btn-danger btn-sm" onclick="modelMappings.splice(${i},1);renderModelMappings()">删除</button></td>
  </tr>`).join('');
  // Inject/refresh datalist once if not present, or update its contents
  let dl=document.getElementById('glm-model-list');
  if(!dl){
    dl=document.createElement('datalist');dl.id='glm-model-list';document.body.appendChild(dl);
  }
  dl.innerHTML=glmOptions;
}
async function loadModelMappings(){
  try{
    const r=await fetch(API+'/admin/api/model-mappings',{headers:authHeaders()});
    const d=await r.json();
    modelMappings=(d.mappings||[]).map(m=>({from:m.from,to:m.to,note:m.note||''}));

  }catch(e){console.error('loadModelMappings failed:',e);toast('加载模型映射失败','error')}
}
async function saveModelMappings(){
  for(let i=0;i<modelMappings.length;i++){
    const m=modelMappings[i];
    if(!m.from||!m.from.trim()){toast(`第 ${i+1} 条映射缺少"客户端模型"`,'error');return}
    if(!m.to||!m.to.trim()){toast(`第 ${i+1} 条映射缺少"GLM 模型"`,'error');return}
  }
  // Check for duplicate from (case-insensitive)
  const seen=new Set();
  for(let i=0;i<modelMappings.length;i++){
    const f=modelMappings[i].from.trim().toLowerCase();
    if(seen.has(f)){toast(`第 ${i+1} 条映射的"客户端模型"重复：${modelMappings[i].from}`,'error');return}
    seen.add(f);
  }
  try{
    const r=await fetch(API+'/admin/api/model-mappings',{method:'PUT',headers:authHeaders(),body:JSON.stringify({mappings:modelMappings})});
    const d=await r.json();
    if(r.ok){
      modelMappings=(d.mappings||[]).map(m=>({from:m.from,to:m.to,note:m.note||''}));
      renderModelMappings();
      toast(`已保存 ${modelMappings.length} 条模型映射`,'success');
    }else{
      toast(d.error||'保存失败','error');
    }
  }catch(e){console.error('saveModelMappings failed:',e);toast('保存失败','error')}
}

async function loadResponsesThinking(){
  try{
    const r=await fetch(API+'/admin/api/responses-thinking',{headers:authHeaders()});
    const d=await r.json();
    responsesThinkingModels=new Set((d.models||[]).map(m=>String(m).toLowerCase()));
    renderResponsesThinkingChips();
    // Sync the custom-input box with any model ids that aren't in the catalog
    // (e.g. operator-added custom ids). Catalog ids are rendered as chips;
    // non-catalog ids appear in the custom input.
    const custom=(d.models||[]).filter(m=>!glmCatalog.some(c=>c.id.toLowerCase()===String(m).toLowerCase()));
    document.getElementById('responsesThinkingCustomInput').value=custom.join(', ');
  }catch(e){console.error('loadResponsesThinking failed:',e);toast('加载 Responses 思考配置失败','error')}
}
function renderResponsesThinkingChips(){
  const wrap=document.getElementById('responsesThinkingChips');
  if(!wrap)return;
  if(!glmCatalog.length){
    wrap.innerHTML='<span class="empty-text" style="color:var(--text-3);font-size:13px;">未加载到 GLM 模型目录，请检查 /admin/api/glm-models 接口</span>';
    return;
  }
  wrap.innerHTML=glmCatalog.map(m=>{
    const on=responsesThinkingModels.has(m.id.toLowerCase());
    const reasoningTag=m.reasoning?'<span class="badge badge-info" style="margin-left:6px;font-size:9px;padding:1px 5px;">reasoning</span>':'';
    const cls=on?'btn btn-success btn-sm':'btn btn-ghost btn-sm';
    const check=on?'<svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3" stroke-linecap="round" stroke-linejoin="round" style="margin-right:4px;"><polyline points="20 6 9 17 4 12"/></svg>':'';
    return `<button type="button" class="${cls}" onclick="toggleResponsesThinking('${escapeHtml(m.id)}')" style="font-family:var(--font-mono);font-size:12px;">${check}${escapeHtml(m.id)}${reasoningTag}</button>`;
  }).join('');
}
function toggleResponsesThinking(modelId){
  const key=modelId.toLowerCase();
  if(responsesThinkingModels.has(key)){
    responsesThinkingModels.delete(key);
  }else{
    responsesThinkingModels.add(key);
  }
  renderResponsesThinkingChips();
}
async function saveResponsesThinking(){
  // Collect: chip-selected ids + custom-input ids (comma-separated).
  const selected=new Set(responsesThinkingModels);
  const customStr=(document.getElementById('responsesThinkingCustomInput').value||'').trim();
  if(customStr){
    for(const part of customStr.split(',')){
      const id=part.trim();
      if(id)selected.add(id.toLowerCase());
    }
  }
  const models=Array.from(selected);
  try{
    const r=await fetch(API+'/admin/api/responses-thinking',{method:'PUT',headers:authHeaders(),body:JSON.stringify({models})});
    const d=await r.json();
    if(r.ok){
      responsesThinkingModels=new Set((d.models||[]).map(m=>String(m).toLowerCase()));
      // Re-sync custom input to only show non-catalog ids
      const custom=(d.models||[]).filter(m=>!glmCatalog.some(c=>c.id.toLowerCase()===String(m).toLowerCase()));
      document.getElementById('responsesThinkingCustomInput').value=custom.join(', ');
      renderResponsesThinkingChips();
      toast(`已保存 Responses 思考配置（${(d.models||[]).length} 个模型）`,'success');
    }else{
      toast(d.error||'保存失败','error');
    }
  }catch(e){console.error('saveResponsesThinking failed:',e);toast('保存失败','error')}
}

    const mu=document.getElementById('modelUsage');
    mu.innerHTML=Object.entries(d.models||{}).map(([m,v])=>`<tr><td class="td-mono">${escapeHtml(m)}</td><td>${v.count}</td><td class="td-mono">${v.avgTtfb}ms</td><td class="td-mono">${v.inputTokens||0}</td><td class="td-mono">${v.tokens}</td></tr>`).join('')
      ||'<tr><td colspan="5"><div class="empty-state"><div class="empty-text">暂无数据</div></div></td></tr>';
    // vceshi0.0.6+: cache byCredential stats + re-render accounts table to
    // refresh usage counts without a full loadAccountsList fetch.
    _cachedByCredential=d.byCredential||{};
    // vceshi0.0.7+: use classList check instead of style.display — pages
    // toggle visibility via the .active class (added/removed by nav clicks),
    // not by setting style.display directly. The old check
    // `style.display!=='none'` was always true (since we never set
    // display:none on the page element), so filterAccounts() ran on every

    const accountsPage=document.getElementById('page-accounts');
    if(accountsPage&&accountsPage.classList.contains('active')&&_cachedAccounts.length>0){
      filterAccounts();
    }
  }catch(e){console.error('loadStats failed:',e)}

function toast(msg,type='success'){
  const container=document.getElementById('toastContainer');
  if(!container)return;
  const el=document.createElement('div');
  el.className='toast toast-'+type;
  const icon=TOAST_ICONS[type]||TOAST_ICONS.info;
  el.innerHTML=`<span class="toast-icon">${icon}</span><span class="toast-msg">${escapeHtml(msg)}</span>`;
  el.addEventListener('click',()=>{
    el.classList.add('leaving');
    setTimeout(()=>el.remove(),200);
  });
  container.appendChild(el);
  setTimeout(()=>{
    if(el.parentNode){
      el.classList.add('leaving');
      setTimeout(()=>el.remove(),200);
    }
  },3000);
}
